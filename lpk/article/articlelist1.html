JavaScript面向对象---继承

function Person(name,age){

this.name='^_'+name;

this.age=age;

}

Person.prototype.showName=function(){

return this.name;

};

Person.prototype.showAge=function(){

return this.age;

};

//如何继承属性？

//父级构造.call(this,arg1,arg2...)

//父级构造.apply(this,arguments);

function Worker(name,age,job){

Person.apply(this,arguments);

this.job=job;

};

//如何继承方法：

//1. 子类.prototype=父类.prototype

//引用:一个修改了，另一个跟着改了

//2. for name in 每次复制一个


//问题： w1 instanceof Person -> false

//3. 子级构造函数.prototype=new 父级构造();

//子级构造函数.prototype.constructor=子级构造函数

Worker.prototype=new Person();

Worker.prototype.constructor=Worker;


Worker.prototype.showJob=function(){

return this.job;

};


var p1=new Person('abc',30);
